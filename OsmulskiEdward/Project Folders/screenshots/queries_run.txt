                                      scheduled_service text, 
                                      gps_code text,

READ STATEMENTS EXAMPLES
SELECT id, name, iso_country, elevation_ft FROM airports WHERE iso_country = 'US' AND elevation_ft <= 50;

SELECT COUNT(*) FROM airports;

SELECT COUNT(*) FROM airports WHERE iso_country = 'US';

SELECT type, COUNT(*) FROM airports WHERE iso_country = 'US' GROUP BY type;

UPDATE STATEMENTS EXAMPLES
SELECT id, type, name FROM airports WHERE id = 6523;

UPDATE airports SET type = 'airport', name = 'Total Rf Airport' WHERE id = 6523;

SELECT id, type, name FROM airports WHERE id = 6523;

UPDATE airports SET type = 'heliport', name = 'Total Rf Heliport' WHERE id = 6523;
